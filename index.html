# 快速开始

# 1、安装

yarn add yzl-ui

# 2、引用组件库

//全部引入
import 'yzl-ui/dist/css/index.css';
import YzlUI from 'yzl-ui';
Vue.use(YzlUI);

//按需引入
import 'yzl-ui/dist/css/demo.css';
import { Demo } from 'yzl-ui';
Vue.use(Demo);
